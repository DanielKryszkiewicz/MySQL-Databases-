SELECT ISSUES.*, ISSUESLISTS.NAME AS LIST_NAME
FROM ISSUES
JOIN ISSUESLISTS ON ISSUES.ISSUESLISTS_ID = ISSUESLISTS.ID;

SELECT ISSUES.*, USERS.FIRSTNAME, USERS.LASTNAME
FROM ISSUES
JOIN USERS ON ISSUES.USER_ID_ASSIGNEDTO = USERS.ID;

SELECT USERS.FIRSTNAME, USERS.LASTNAME, COUNT(ISSUES.ID) AS NUMBER_OF_TASKS
FROM USERS
JOIN ISSUES ON USERS.ID = ISSUES.USER_ID_ASSIGNEDTO
GROUP BY USERS.ID;
